from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy.orm import Session
from sqlalchemy import func
from typing import List, Optional
from app.database import get_db
from app.models.models import Bank, Product
from pydantic import BaseModel
from datetime import datetime

router = APIRouter()

class BankSchema(BaseModel):
    id: int
    name: str
    name_tj: Optional[str]
    short_name: Optional[str]
    website: Optional[str]
    logo_url: Optional[str]
    phone: Optional[str]
    address: Optional[str]
    type: str
    is_active: bool
    capital: Optional[float]
    assets: Optional[float]
    founded_year: Optional[int]
    rating_score: Optional[float]
    products_count: Optional[int] = 0

    class Config:
        from_attributes = True

@router.get("/", response_model=List[BankSchema])
def get_banks(
    type: Optional[str] = Query(None, description="bank / mfo / nko"),
    is_active: bool = True,
    db: Session = Depends(get_db)
):
    query = db.query(Bank).filter(Bank.is_active == is_active)
    if type:
        query = query.filter(Bank.type == type)
    banks = query.order_by(Bank.name).all()
    result = []
    for bank in banks:
        count = db.query(func.count(Product.id)).filter(
            Product.bank_id == bank.id,
            Product.is_active == True
        ).scalar()
        b = BankSchema.from_orm(bank)
        b.products_count = count
        result.append(b)
    return result

@router.get("/{bank_id}", response_model=BankSchema)
def get_bank(bank_id: int, db: Session = Depends(get_db)):
    bank = db.query(Bank).filter(Bank.id == bank_id).first()
    if not bank:
        raise HTTPException(status_code=404, detail="Банк не найден")
    count = db.query(func.count(Product.id)).filter(
        Product.bank_id == bank.id,
        Product.is_active == True
    ).scalar()
    b = BankSchema.from_orm(bank)
    b.products_count = count
    return b
